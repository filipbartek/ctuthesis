\subsection{Machine learning}

\begin{frame}{Heuristics in theorem proving}
How should we select the next clause to activate?
\pause
\medskip

Heuristics:
\begin{itemize}
\item Age: Prioritize clauses derived early
\item Weight: Prioritize small clauses
\end{itemize}

Can we do better?
\end{frame}

\begin{frame}{Machine learning (ML) for theorem proving}
Challenge: Learn \enquote{intuition} from training data
\pause
\bigskip

\begin{block}{Supervised ML}
\begin{enumerate}
\item Gather data: Run the prover on the available problems
%\item Split the problems into train and test set (possibly repeatedly)
\item Train a ML model on training problems
\item Evaluate the model on test problems
\end{enumerate}
\end{block}
